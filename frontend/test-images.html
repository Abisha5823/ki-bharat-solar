// Check if files exist
async function checkFile(url) {
    try {
        const response = await fetch(url, { method: 'HEAD' });
        console.log(`${url}: ${response.ok ? '✅ EXISTS' : '❌ NOT FOUND'}`);
    } catch(e) {
        console.log(`${url}: ❌ ERROR`);
    }
}
